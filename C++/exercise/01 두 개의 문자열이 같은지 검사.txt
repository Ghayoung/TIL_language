// ConsoleApplication2.cpp : 콘솔 응용 프로그램에 대한 진입점을 정의합니다.
//

#include "stdafx.h"
#include<iostream>
#include<string>
using namespace std;


int main()
{
	string pw;
	string pw2;

	while (true) {
		cout << "새 암호를 입력하세요: ";
		getline(cin, pw);
		cout << "\n새 암호를 다시 입력하세요: ";
		getline(cin, pw2);

		if (pw == pw2) {
			cout << "\n변경되었습니다.\n";
			break;
		}
		else
			cout << "\n다시 입력해 주세요.\n\n";
	}

    return 0;
}
